59 class BaseControllerFixed(threading.Thread):
60  """
61  This Class can be used to simulate floating base robots that
62  have an under-actuated base (e.g. like quadrupeds, mobile robots)
63 
64  ...
65 
66  Attributes
67  ----------
68  q, q_des : numpy array
69  actual /desired joint positions
70  qd, qd_des : numpy array
71  actual /desired joint velocities
72  tau, tau_ffwd : numpy array
73  actual /feed-forward torques
74  x_ee : numpy array
75  position of the end-effector expressed in the base_link frame
76  contactForceW : numpy array
77  contact force at the end-effector expressed in the world frame
78  contactMomentW : numpy array
79  contact moment at the end-effector expressed in the world frame
80 
81  Methods
82  -------
83  loadModelAndPublishers(xacro_path=None)
84  loads publishers for visual features (ros_pub), joint commands and declares subscriber to /ground_truth and /joint_states
85  startSimulator(world_name = None)
86  Starts gazebo simulator with ros_impedance_controller
87  send_des_jstate(self, q_des, qd_des, tau_ffwd)
88  publishes /command topic with set-points for joint positions, velocities and feed-forward torques
89  startupProcedure():
90  initialize PD gains
91  initVars()
92  initializes class variables
93  logData()
94  fill in the X_log variables for plotting purposes, it needs to be called at every loop
95  _receive_jstate(msg)
96  callback associated to the joint state subscriber, fills in q, qd, tau arrays
97 
98  """
